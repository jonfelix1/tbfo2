============
BASE GRAMMAR
============
//PER LINE STATEMENT
S -> S NEXTLINE | EXPRCONT NEXTLINE | EXPRCONT | CRASH COMMENTCONT
S -> EXPRCONT NEXTLINE
NEXTLINE -> ENDLINE S | ENDLINE EXPRCONT | ENDLINE PASS | 'ENDLINE'

//EXPRESSIONS
EXPRCONT -> 'FLOAT' | 'INT' | IDENTIFIER' | 'STRING' | 'MULTILINECOMMENT' | 'FALSE' | 'TRUE' | 'NONE'

EXPRCONT -> LEFTP RIGHTCONT | ATOM OPERATION2 | EXPRCONT OPERATION2 | IDENTIFIER FUNCTIONPAR | IDENTIFIER DOTFUNCT | IDENTIFIER BRACKDOTFUNCT | STRING DOTFUNCT | LEFTSB RIGHTFORSBCONT | IDENTIFIER FOREXPR2

RIGHTFORSBCONT -> FORSBCONT RIGHTSB
FORSBCONT -> IDENTIFIER FORSBCONT1
FORSBCONT1 -> FOR FOREXPR1
BRACKDOTFUNCT -> BRACKETS DOTFUNCT
DOTFUNCT -> PERIOD DOTFUNCT2
DOTFUNCT2 -> IDENTIFIER FUNCTIONPAR
DOTFUNCT2 -> 'IDENTIFIER'
EXPRCONT -> IDENTIFIER BRACKETS
BRACKETS -> LEFTSB RIGHTSBCONT
EXPRCONT -> LEFTSB RIGHTSBCONT
RIGHTSBCONT -> SBCONT RIGHTSB | RIGHTSBCONT -> EXPRCONT RIGHTSB
SBCONT -> EXPRCONT NEXTSBCONT
NEXTSBCONT -> COMMA SBCONT 


OPERATION2 -> COMPARATOR ATOM | COMPARATOR EXPRCONT | COMPAREEQUAL ATOM | COMPAREEQUAL EXPRCONT | COMPAREEQUALITY ATOM | COMPAREEQUALITY EXPRCONT | OPERATOR ATOM | LOGICALOPERATOR ATOM | OPERATOR EXPRCONT | LOGICALOPERATOR EXPRCONT | COMMA ATOM | COMMA EXPRCONT

RIGHTCONT -> EXPRCONT RIGHTP

//FILENAME
FILENAME -> ATOM FILENAMEFORMAT
FILENAMEFORMAT -> PERIOD IDENTIFIER

//ASSIGNMENT
S -> IDENTIFIER ASSIGNMENT
ASSIGNMENT -> EQUALS EXPRCONT


S -> 'MAGICMETHOD'
S -> IDENTIFIER FUNCTIONPAR

//FUNCTIONPAR
FUNCTIONPAR -> LEFTP FUNCTIONARG
FUNCTIONPAR -> LEFTP RIGHTP
FUNCTIONARG -> EXPRCONT RIGHTP

////////////////
IMPORT
/////////
S -> FROM IMP1 | IMPORT IMP3
IMP1 -> IDENTIFIER IMP2
IMP2 -> IMPORT IMP3
IMP3 -> IDENTIFIER | IDENTIFIER IMP4
IMP4 -> AS IDENTIFIER

///////////
RAISE
//////////
S -> RAISE RAISEARG
RAISEARG -> IDENTIFIER FUNCTIONPAR

/////////
IF
/////////
// FIRST CASE
S -> IF IF1
IF1 -> EXPRCONT IF2
IF2 -> COLON NEXTLINE

//SECOND CASE
S -> IF00 ELSEINIF
IF00 -> IF IF1
IF04 -> COLON NEXTLINE
ELSEINIF -> ELSE IF04

//THIRD CASE
S -> IF000 ELIFINIF
IF000 -> IF IF1
ELIFINIF -> ELIF IF004
IF004 -> EXPRCONT IF005
IF005 -> COLON IF006
IF006 -> NEXTLINE ELIFINIF | NEXTLINE

//FOURTH CASE
S -> IF IF0000
IF0000 -> IF1 IF0002
IF0002 -> ELIFINIF ELSEINIF

///////
RETURN
//////
S -> RETURN IDENTIFIER

/////
DEF
////
S -> DEFMETHOD NEXTLINE
DEFMETHOD -> DEF DEFMETHOD1
DEFMETHOD1 -> DEFMETHOD2 COLON
DEFMETHOD2 -> DEFMETHOD4 DEFRETURNTYPE | IDENTIFIER DEFMETHOD3 
DEFMETHOD3 -> LEFTP RIGHTPAR | LEFTP RIGHTP
DEFMETHOD4 -> IDENTIFIER DEFMETHOD3 
DEFRETURNTYPE -> ARROW IDENTIFIER
RIGHTPAR -> PAR RIGHTP
NEXTPAR -> COMMA PAR
PAR -> PAR NEXTPAR | 'IDENTIFIER' | IDENTIFIER PAR2 | IDENTIFIER PAR5 | PAR4 PAR5
PAR2 -> COLON IDENTIFIER
PAR4 -> IDENTIFIER PAR2
PAR5 -> EQUALS EXPRCONT

/////
CLASS
/////
S -> CLASSDECLARE NEXTLINE
CLASSDECLARE -> CLASS CLASSDECLARE1
CLASSDECLARE1 -> IDENTIFIER COLON
CLASSDECLARE1 -> CLASSDECLARE2 COLON
CLASSDECLARE2 -> IDENTIFIER CLASSPAR
CLASSPAR -> LEFTP RIGHTCLASSPAR
RIGHTCLASSPAR -> IDENTIFIER RIGHTP

////
WITH
//// 
S -> WITHDECLARE NEXTLINE
WITHDECLARE -> WITH WITHDECLARE1
WITHDECLARE1 -> IDENTIFIER WITHDECLARE2
WITHDECLARE2 -> FUNCTIONPAR WITHDECLARE3
WITHDECLARE2 -> DOTFUNCT WITHDECLARE3
WITHDECLARE3 -> AS WITHDECLARE4
WITHDECLARE4 -> IDENTIFIER COLON

/////
FOR
/////
S -> FOREXPR NEXTFORSTATEMENT
FOREXPR -> FOR FOREXPR0
FOREXPR0 -> FOREXPR1 COLON
FOREXPR1 -> IDENTIFIER FOREXPR2 | IDENTIFIER FOREXPR11
FOREXPR11 -> COMMA FOREXPR12
FOREXPR12 -> IDENTIFIER FOREXPR13
FOREXPR13 -> IN DICT
FOREXPR2 -> IN FOREXPR3 | IN DICT
FOREXPR3 -> 'IDENTIFIER' | RANGE FOREXPR4
FOREXPR4 -> LEFTP FOREXPR5
FOREXPR5 -> INT FOREXPR6 | INT RIGHTP
FOREXPR6 -> COMMA FOREXPR7
FOREXPR7 -> INT FOREXPR8 | INT RIGHTP
FOREXPR8 -> COMMA FOREXPR9
FOREXPR9 -> INT RIGHTP   
NEXTFORSTATEMENT -> ENDLINE FORSTATEMENT | ENDLINE S
FORSTATEMENT -> FORSTATEMENT NEXTFORSTATEMENT | 'BREAK' | 'CONTINUE' | IF IFFOR1
IFFOR1 -> EXPRCONT IFFOR2
IFFOR2 -> COLON NEXTFORSTATEMENT
FORSTATEMENT -> IFFOR00 ELSEINIFFOR
IFFOR00 -> IF IFFOR1
IFFOR04 -> COLON NEXTFORSTATEMENT
ELSEINIFFOR -> ELSE IFFOR04
FORSTATEMENT -> IFFOR000 ELIFINIFFOR
IFFOR000 -> IF IFFOR1
ELIFINIFFOR -> ELIF IFFOR004
IFFOR004 -> EXPRCONT IFFOR005
IFFOR005 -> COLON IFFOR006
IFFOR006 -> NEXTFORSTATEMENT ELIFINIFFOR | NEXTFORSTATEMENT
FORSTATEMENT -> IF IFFOR0000
IFFOR0000 -> IFFOR1 IFFOR0002
IFFOR0002 -> ELIFINIFFOR ELSEINIFFOR

/////
DICT
/////
DICT -> LEFTCB RIGHTDICTCONT
RIGHTDICTCONT -> DICTEXPR RIGHTCB
DICTEXPR -> LEFTP KEYVAL
KEYVAL -> KEY KEYVAL2
KEYVAL2 -> COMMA KEYVAL3
KEYVAL3 -> VALUE RIGHTP | VALUE KEYVAL4
KEYVAL4 -> RIGHTP KEYVAL5
KEYVAL5 -> COMMA DICTEXPR
KEY -> 'FLOAT' | 'INT' | 'STRING'
VALUE -> 'FLOAT' | 'INT' | 'STRING'

//////
WHILE
/////
S -> WHILEEXPR NEXTFORSTATEMENT
WHILEEXPR -> WHILE WHILEEXPR1
WHILEEXPR1 -> EXPRCONT COLON
